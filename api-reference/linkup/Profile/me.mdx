---
title: 'Get My Profile'
api: 'POST /v1/profile/me'
description: 'Get current user\'s LinkedIn profile information'
---

Get current user's LinkedIn profile information. This endpoint retrieves detailed information about your own LinkedIn profile.

### Header Parameters

<ParamField header="x-api-key" type="string" required>
  Your API key
</ParamField>

### Body Parameters

<ParamField body="login_cookie" type="string" required>
  LinkedIn authentication cookie obtained from the login/verify process
</ParamField>

### Response

<ResponseField name="profile" type="object">
  <Expandable title="Toggle Profile Object">
    <ResponseField name="basic_info" type="object">
      <Expandable title="Toggle Basic Info">
        <ResponseField name="full_name" type="string">Full name of the user</ResponseField>
        <ResponseField name="headline" type="string">Professional headline</ResponseField>
        <ResponseField name="location" type="string">Location information</ResponseField>
        <ResponseField name="profile_picture" type="string">URL of the profile picture</ResponseField>
        <ResponseField name="connection_count" type="integer">Number of connections</ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="experience" type="array">
      <Expandable title="Toggle Experience">
        <ResponseField name="title" type="string">Job title</ResponseField>
        <ResponseField name="company" type="string">Company name</ResponseField>
        <ResponseField name="location" type="string">Job location</ResponseField>
        <ResponseField name="start_date" type="string">Start date of position</ResponseField>
        <ResponseField name="end_date" type="string">End date of position (null if current)</ResponseField>
        <ResponseField name="description" type="string">Job description</ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="education" type="array">
      <Expandable title="Toggle Education">
        <ResponseField name="school" type="string">School name</ResponseField>
        <ResponseField name="degree" type="string">Degree obtained</ResponseField>
        <ResponseField name="field" type="string">Field of study</ResponseField>
        <ResponseField name="start_year" type="integer">Start year</ResponseField>
        <ResponseField name="end_year" type="integer">End year</ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="skills" type="array">
      Array of professional skills
    </ResponseField>
  </Expandable>
</ResponseField>
