
---
title: 'Login'
api: 'POST /v1/linkup/auth/login'
description: 'Authenticate with LinkedIn credentials'
---

Authenticate with your LinkedIn account. If verification is required, a security code will be sent to your email.

### Header Parameters

<ParamField header="x-api-key" type="string" required>
  Your API key
</ParamField>

### Body Parameters

<ParamField body="email" type="string" required>
  Your LinkedIn email address
</ParamField>

<ParamField body="password" type="string" required>
  Your LinkedIn password
</ParamField>

### Response

<ResponseField name="status" type="string">
  Authentication status
</ResponseField>

<ResponseField name="requires_verification" type="boolean">
  Whether security code verification is needed
</ResponseField>

<ResponseField name="cookie" type="string">
  LinkedIn authentication cookie (if no verification required)
</ResponseField>

<TabItem value="go">

```go
package main

import (
	"bytes"
	"fmt"
	"net/http"
	"net/url"
)

func main() {
	apiURL := "https://api.linkup.com/v1/linkup/auth/login"
	data := url.Values{}
	data.Set("email", "user@example.com")
	data.Set("password", "your-password")

	req, err := http.NewRequest("POST", apiURL, bytes.NewBufferString(data.Encode()))
	if err != nil {
		fmt.Println("Error creating request:", err)
		return
	}

	req.Header.Set("x-api-key", "your-api-key")
	req.Header.Set("Content-Type", "application/x-www-form-urlencoded")

	client := &http.Client{}
	resp, err := client.Do(req)
	if err != nil {
		fmt.Println("Error making request:", err)
		return
	}
	defer resp.Body.Close()

	fmt.Println("Response Status:", resp.Status)
}
```

<TabItem value="java">

```java
import java.io.OutputStream;
import java.net.HttpURLConnection;
import java.net.URL;

public class Main {
    public static void main(String[] args) {
        try {
            URL url = new URL("https://api.linkup.com/v1/linkup/auth/login");
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("POST");
            conn.setRequestProperty("x-api-key", "your-api-key");
            conn.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
            conn.setDoOutput(true);

            String data = "email=user@example.com&password=your-password";

            try (OutputStream os = conn.getOutputStream()) {
                os.write(data.getBytes());
                os.flush();
            }

            int responseCode = conn.getResponseCode();
            System.out.println("Response Code: " + responseCode);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```
