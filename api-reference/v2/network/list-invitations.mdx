---
title: 'List Invitations'
api: 'POST /v2/network'
description: 'Retrieve a paginated list of pending invitations received.'
---

## Overview

Returns a paginated list of pending connection invitations received by the account. Pagination is **offset-based** — set `count` for the page size and `offset` to skip ahead.

<Note>This endpoint consumes **1 credit per 50 results** returned. LinkedIn returns up to **100 invitations per internal request**, so larger `count` values trigger multiple upstream calls automatically.</Note>

## Request

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication.
</ParamField>

<ParamField body="account_id" type="string" required>
  The unique identifier of the account to use.
</ParamField>

<ParamField body="action" type="string" required>
  Must be `"list_invitations"`.
</ParamField>

<ParamField body="params" type="object">
  <Expandable title="params properties">
    <ParamField body="count" type="integer" default={10}>
      Number of invitations to return. Defaults to `10`. **Capped at `1000`** per call — to retrieve more, paginate with `offset`.
    </ParamField>
    <ParamField body="offset" type="integer" default={0}>
      Zero-indexed offset into the invitations list. Defaults to `0`. Use `pagination.next_offset` from a previous response to fetch the next page.
    </ParamField>
    <ParamField body="invitation_type" type="string">
      Filter by invitation type (e.g. `"CONNECTION"`). Omit for all types.
    </ParamField>
  </Expandable>
</ParamField>


## Response

<ResponseField name="success" type="boolean">
  Indicates whether the request was successful.
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="data properties">
    <ResponseField name="invitations" type="array">
      List of invitation objects with `entity_urn`, `shared_secret`, sender info, type, and timing.
    </ResponseField>
    <ResponseField name="total_results" type="integer">
      Number of invitations returned in this response.
    </ResponseField>
    <ResponseField name="total_available_results" type="integer">
      Total number of pending invitations available on the account.
    </ResponseField>
    <ResponseField name="pagination" type="object">
      <Expandable title="pagination properties">
        <ResponseField name="offset" type="integer">Offset of the first invitation in this response.</ResponseField>
        <ResponseField name="count" type="integer">Number of invitations returned (mirrors `total_results`).</ResponseField>
        <ResponseField name="next_offset" type="integer">Offset to send as `offset` in the next request. `null` when there are no more results.</ResponseField>
        <ResponseField name="has_more" type="boolean">`true` if more invitations are available beyond this batch.</ResponseField>
        <ResponseField name="results_per_page" type="integer">Maximum batch size used per upstream request.</ResponseField>
        <ResponseField name="pages_fetched" type="integer">Number of internal upstream batches consumed to assemble this response.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="metadata" type="object">
  <Expandable title="metadata properties">
    <ResponseField name="action" type="string">The action that was performed (`"list_invitations"`).</ResponseField>
    <ResponseField name="account_id" type="string">The account ID used for the request.</ResponseField>
    <ResponseField name="credits_consumed" type="integer">Number of credits consumed (1 per 50 results).</ResponseField>
    <ResponseField name="timestamp" type="string">ISO 8601 timestamp of the response.</ResponseField>
  </Expandable>
</ResponseField>
