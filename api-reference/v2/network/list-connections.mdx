---
title: 'List Connections'
api: 'POST /v2/network'
description: 'Retrieve a paginated list of connections.'
---

## Overview

Returns a paginated list of the account's connections. Pagination is **offset-based** (LinkedIn provides a stable, addressable list — no opaque cursor needed). Set `count` for the page size and `offset` to skip ahead.

<Note>This endpoint consumes **1 credit per 50 results** returned. LinkedIn returns up to **100 connections per internal request**, so requesting `count: 250` triggers ~3 upstream calls automatically.</Note>

## Request

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication.
</ParamField>

<ParamField body="account_id" type="string" required>
  The unique identifier of the account to use.
</ParamField>

<ParamField body="action" type="string" required>
  Must be `"list_connections"`.
</ParamField>

<ParamField body="params" type="object">
  <Expandable title="params properties">
    <ParamField body="count" type="integer" default={10}>
      Number of connections to return. Defaults to `10`. **Capped at `1000`** per call — to retrieve more, paginate with `offset`.
    </ParamField>
    <ParamField body="offset" type="integer" default={0}>
      Zero-indexed offset into the connections list. Defaults to `0`. Use `pagination.next_offset` from a previous response to fetch the next page.
    </ParamField>
  </Expandable>
</ParamField>


## Response

<ResponseField name="success" type="boolean">
  Indicates whether the request was successful.
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="data properties">
    <ResponseField name="connections" type="array">
      List of connection objects.
      <Expandable title="connection properties">
        <ResponseField name="name" type="string">Display name of the connection.</ResponseField>
        <ResponseField name="job_title" type="string">LinkedIn headline / current job title.</ResponseField>
        <ResponseField name="profile_url" type="string">LinkedIn profile URL.</ResponseField>
        <ResponseField name="profile_picture" type="string">URL of the connection's profile picture.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="total_results" type="integer">
      Number of connections returned in this response.
    </ResponseField>
    <ResponseField name="total_available_results" type="integer">
      Total number of connections available in the account.
    </ResponseField>
    <ResponseField name="pagination" type="object">
      <Expandable title="pagination properties">
        <ResponseField name="offset" type="integer">Offset of the first connection in this response.</ResponseField>
        <ResponseField name="count" type="integer">Number of connections returned (mirrors `total_results`).</ResponseField>
        <ResponseField name="next_offset" type="integer">Offset to send as `offset` in the next request to fetch the next page. `null` when there are no more results.</ResponseField>
        <ResponseField name="has_more" type="boolean">`true` if more connections are available beyond this batch.</ResponseField>
        <ResponseField name="results_per_page" type="integer">Maximum batch size used per upstream request.</ResponseField>
        <ResponseField name="pages_fetched" type="integer">Number of internal upstream batches consumed to assemble this response.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="metadata" type="object">
  <Expandable title="metadata properties">
    <ResponseField name="action" type="string">
      The action that was performed (`"list_connections"`).
    </ResponseField>
    <ResponseField name="account_id" type="string">
      The account ID used for the request.
    </ResponseField>
    <ResponseField name="credits_consumed" type="integer">
      Number of credits consumed (1 per 50 results).
    </ResponseField>
    <ResponseField name="timestamp" type="string">
      ISO 8601 timestamp of the response.
    </ResponseField>
  </Expandable>
</ResponseField>
